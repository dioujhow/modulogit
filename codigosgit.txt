git push -u origin master							enviar o repositorio local para o repositorio remoto
git log												ver todos os commits 
git diff											ver o que foi alterado ou modificado de todos arquivos
git diff -- "nome do arquivo"  						ver o que foi alterado ou modificado no arquivo
git status											ver os arquivos que foram alterados ou adicionados
git reset --solf/hard  " colocar o codigo do commit"						voltar a um commit anterior
 git add -A  										adicionar todos arquivos que foram modificados ou adicionados
git commit -am "comentar o que foi feito"
